### 步骤 3: 生成 run-tests.sh

**位置**：`{repo_path}/run-tests.sh`（仓库根目录）

**目标**：一键脚本，编译 + 仿真 + 解析结果

**脚本关键要求**：

1. **⚠️ CRITICAL: 不要使用 `set -e`**：必须允许测试失败后继续执行统计
2. **⚠️ CRITICAL: 不要使用绝对路径**：所有路径必须是相对路径
3. **🔴 CRITICAL: 必须生成 result.json**：包含 passed, failed, error, total, pass_rate, test_framework, exit_code
4. **🔴 CRITICAL: 必须输出标准格式统计**：Passed/Failed/Error/Total（向后兼容）

**脚本模板**：

```bash
#!/bin/bash
# Auto-generated test runner for Verilog/SystemVerilog
# Repository: {repo_url}

echo "=========================================="
echo "Running Verilog/SystemVerilog Tests"
echo "=========================================="

SCRIPT_DIR="$(cd "$(dirname "${{BASH_SOURCE[0]}}" )" && pwd)"
cd "$SCRIPT_DIR"

echo "Working directory: $SCRIPT_DIR"
echo "Current git status:"
git log -1 --oneline 2>/dev/null || echo "Not a git repository"
echo ""

# 初始化统计变量
TOTAL_PASSED=0
TOTAL_FAILED=0
TOTAL_ERROR=0
OVERALL_EXIT_CODE=0
TEST_OUTPUT_FILE="./test_output_$$.txt"
TEST_FRAMEWORK="unknown"

# ============================================
# 🔴 Step 0: Environment Check
# ============================================
echo "=========================================="
echo "Step 0: Environment Check"
echo "=========================================="

echo "Checking simulation tools..."
which iverilog >/dev/null 2>&1 && echo "  ✓ iverilog found: $(iverilog -V 2>&1 | head -1)" || echo "  ✗ iverilog NOT found"
which verilator >/dev/null 2>&1 && echo "  ✓ verilator found: $(verilator --version 2>&1 | head -1)" || echo "  - verilator not found"
which cocotb-config >/dev/null 2>&1 && echo "  ✓ cocotb found: $(cocotb-config --version 2>&1)" || echo "  - cocotb not found"
python3 -c "import cocotb" 2>/dev/null && echo "  ✓ cocotb Python module available" || echo "  - cocotb Python module not available"
echo ""

# ============================================
# Step 1: Compile (iverilog) or Configure (cocotb)
# ============================================
echo "=========================================="
echo "Step 1: Compile / Configure"
echo "=========================================="

# TODO: 根据检测到的仿真工具生成编译命令
# 例如 iverilog:
# iverilog -g2012 -o sim.vvp -I include/ src/*.v tb/*.v
# 例如 cocotb:
# cd tests && make SIM=icarus

# ============================================
# Step 2: Run Simulation
# ============================================
echo ""
echo "=========================================="
echo "Step 2: Run Simulation"
echo "=========================================="

rm -f "$TEST_OUTPUT_FILE"

# TODO: 替换为实际的仿真命令
# 例如 iverilog: vvp sim.vvp 2>&1 | tee "$TEST_OUTPUT_FILE"
# 例如 cocotb: make SIM=icarus 2>&1 | tee "$TEST_OUTPUT_FILE"

TEST_EXIT_CODE=${{PIPESTATUS[0]}}

# ============================================
# Step 3: Parse Results
# ============================================

# --- cocotb 结果解析 ---
parse_cocotb_results() {{
    local output_file="$1"
    # cocotb 输出: "X passed, Y failed"
    RUN_PASSED=$(grep -oP '\d+(?= passed)' "$output_file" 2>/dev/null | tail -1 || echo "0")
    RUN_FAILED=$(grep -oP '\d+(?= failed)' "$output_file" 2>/dev/null | tail -1 || echo "0")
    RUN_TOTAL=$((RUN_PASSED + RUN_FAILED))
    TEST_FRAMEWORK="cocotb"

    # 也检查 results.xml (JUnit format)
    if [ -f "results.xml" ] && [ "$RUN_TOTAL" -eq 0 ]; then
        RUN_TOTAL=$(grep -c '<testcase' results.xml 2>/dev/null || echo "0")
        RUN_FAILED=$(grep -c '<failure' results.xml 2>/dev/null || echo "0")
        RUN_PASSED=$((RUN_TOTAL - RUN_FAILED))
    fi
}}

# --- iverilog/vvp 结果解析 ---
parse_iverilog_results() {{
    local output_file="$1"
    RUN_PASSED=0
    RUN_FAILED=0

    # 检查 $display 输出中的 PASS/FAIL
    while IFS= read -r line; do
        line_upper=$(echo "$line" | tr '[:lower:]' '[:upper:]')
        if echo "$line_upper" | grep -qE '\bPASS(ED)?\b' && ! echo "$line_upper" | grep -q 'PASSWORD'; then
            RUN_PASSED=$((RUN_PASSED + 1))
        fi
        if echo "$line_upper" | grep -qE '\bFAIL(ED)?\b'; then
            RUN_FAILED=$((RUN_FAILED + 1))
        fi
    done < "$output_file"

    # 检查 "All tests passed" 模式
    if grep -qi "all tests\? pass" "$output_file" 2>/dev/null; then
        if [ "$RUN_PASSED" -eq 0 ]; then RUN_PASSED=1; fi
        RUN_FAILED=0
    fi

    RUN_TOTAL=$((RUN_PASSED + RUN_FAILED))

    # 如果没有 PASS/FAIL 输出，检查 $finish（正常结束=通过）
    if [ "$RUN_TOTAL" -eq 0 ]; then
        if grep -q '\$finish\|finish called' "$output_file" 2>/dev/null; then
            RUN_PASSED=1
            RUN_TOTAL=1
        fi
    fi

    TEST_FRAMEWORK="iverilog"
}}

# --- UVM 结果解析 ---
parse_uvm_results() {{
    local output_file="$1"
    UVM_ERRORS=$(grep -oP 'UVM_ERROR\s*:\s*\K\d+' "$output_file" 2>/dev/null | tail -1 || echo "0")
    UVM_FATALS=$(grep -oP 'UVM_FATAL\s*:\s*\K\d+' "$output_file" 2>/dev/null | tail -1 || echo "0")

    if [ "$((UVM_ERRORS + UVM_FATALS))" -eq 0 ]; then
        RUN_PASSED=1
        RUN_FAILED=0
    else
        RUN_PASSED=0
        RUN_FAILED=1
    fi
    RUN_TOTAL=1
    TEST_FRAMEWORK="uvm"
}}

# 根据实际测试框架调用解析函数
if [ -f "$TEST_OUTPUT_FILE" ]; then
    # TODO: 根据实际测试框架选择解析函数
    # 优先 cocotb → UVM → iverilog
    if grep -q "cocotb\|passed.*failed" "$TEST_OUTPUT_FILE" 2>/dev/null; then
        parse_cocotb_results "$TEST_OUTPUT_FILE"
    elif grep -q "UVM_ERROR\|UVM_FATAL\|UVM_INFO" "$TEST_OUTPUT_FILE" 2>/dev/null; then
        parse_uvm_results "$TEST_OUTPUT_FILE"
    else
        parse_iverilog_results "$TEST_OUTPUT_FILE"
    fi

    TOTAL_PASSED=$((TOTAL_PASSED + ${{RUN_PASSED:-0}}))
    TOTAL_FAILED=$((TOTAL_FAILED + ${{RUN_FAILED:-0}}))
fi

if [ $TEST_EXIT_CODE -ne 0 ]; then
    OVERALL_EXIT_CODE=1
fi

# ============================================
# Step 4: Output Results
# ============================================
echo ""
echo "=========================================="
echo "Test Results Summary:"
echo "=========================================="

TOTAL=$((TOTAL_PASSED + TOTAL_FAILED + TOTAL_ERROR))

echo "Passed: $TOTAL_PASSED"
echo "Failed: $TOTAL_FAILED"
echo "Error: $TOTAL_ERROR"
echo "Total: $TOTAL"

if [ $TOTAL -gt 0 ]; then
    PASS_RATE=$(awk "BEGIN {{printf \"%.1f\", ($TOTAL_PASSED/$TOTAL)*100}}")
    echo "Pass Rate: $PASS_RATE%"
else
    echo "Pass Rate: 0.0%"
fi

echo "Exit Code: $OVERALL_EXIT_CODE"
echo "=========================================="

# 🆕 生成 result.json
if [ $TOTAL -gt 0 ]; then
    PASS_RATE_JSON=$(awk "BEGIN {{printf \"%.3f\", ($TOTAL_PASSED/$TOTAL)}}")
else
    PASS_RATE_JSON="0.0"
fi

cat > "$SCRIPT_DIR/result.json" << RESULT_EOF
{{
  "passed": $TOTAL_PASSED,
  "failed": $TOTAL_FAILED,
  "error": $TOTAL_ERROR,
  "total": $TOTAL,
  "pass_rate": $PASS_RATE_JSON,
  "test_framework": "$TEST_FRAMEWORK",
  "exit_code": $OVERALL_EXIT_CODE
}}
RESULT_EOF

echo ""
echo "✓ Generated result.json"

rm -f "$TEST_OUTPUT_FILE"

exit $OVERALL_EXIT_CODE
```

**关键点**：
1. **不使用 `set -e`**: 允许测试失败后继续统计
2. **不使用绝对路径**: 保证可移植性
3. **必须生成 result.json**: 供自动化系统解析
4. **根据项目实际使用的仿真工具写对应的命令和解析函数**
5. **包含 cocotb / iverilog / UVM 三种解析器**
6. **脚本在仓库根目录运行**
